import React from 'react';
import { useAuth } from '../context/AuthContext';
import { Button } from './ui/button';
import { AlertTriangle, UserCheck } from 'lucide-react';

export const ImpersonationBanner = () => {
  const { isImpersonating, user, stopImpersonating } = useAuth();

  // This component renders nothing if not in impersonation mode
  if (!isImpersonating) {
    return null;
  }

  return (
    <div 
      className="bg-yellow-400 text-yellow-900 font-semibold p-2 fixed top-0 w-full z-[100] flex items-center justify-center shadow-lg"
      style={{
        // Adding a slight RTL adjustment for the Persian UI
        flexDirection: document.dir === 'rtl' ? 'row-reverse' : 'row'
      }}
    >
      <AlertTriangle className="w-5 h-5 mx-3 flex-shrink-0" />
      <div className="text-center text-sm">
        <span className="hidden sm:inline">You are currently impersonating </span>
        <strong>{user?.full_name || user?.email}</strong>.
      </div>
      <Button
        variant="outline"
        size="sm"
        className="mx-4 bg-white text-yellow-800 hover:bg-yellow-50 border-yellow-500 hover:border-yellow-600"
        onClick={stopImpersonating}
      >
        <UserCheck className="w-4 h-4 mr-2" />
        Stop Impersonating
      </Button>
    </div>
  );
};
