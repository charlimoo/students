export interface AcademicRecord {
  id: string;
  degree: string;
  country: string;
  university: string;
  field: string;
  gpa: string;
  documentFile: File | null;
}

export interface UniversityProgramChoice {
  id: string;
  priority: number;
  university: string;
  field: string;
}

export interface DocumentUpload {
  id: string;
  documentType: string;
  file: File | null;
}

export interface AdmissionFormData {
  // Section 1: Personal Info
  fullName: string;
  fatherName: string;
  grandfatherName: string;
  birthDate: string;
  nationality: string;
  email: string;
  
  // Section 2: Academic History
  academicRecords: AcademicRecord[];
  
  // Section 3: University & Program
  universityPrograms: UniversityProgramChoice[];
  
  // Section 4: Document Uploads
  documentUploads: DocumentUpload[];
  
  // This is a simplified structure. Other fields from the Persian form are omitted for clarity
  // but would be included here in a full implementation.
  degreeType?: string;
  entryYear?: string;
  motivationLetter?: string;
  previousExperience?: string;
}

export interface AdmissionFormProps {
  onBack: () => void;
  onSubmit: () => void;
}

export interface FormStepProps {
  formData: AdmissionFormData;
  onInputChange: (key: string, value: any) => void;
  onFileUpload: (key: string, file: File | null) => void;
}
